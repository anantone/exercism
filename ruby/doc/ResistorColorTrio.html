def label
  zeros = '0' * self.band3
  value = self.band1.to_s + self.band2.to_s + zeros
  if value.to_i < 1000
    value << ' ohms'
  else
    value.gsub!(/000\b/, ' kiloohms')
  end
  'Resistor value: %s' % value
end
