def self.pangram?(sentence)
  count = 0
  unless sentence == ""
    alphabet = "abcdefghijklmnopqrstuvwxyz"
    alphabet.each_char { |char|  count += 1 if sentence.downcase.include?(char)}
  end
  count == 26 ? true : false
end
