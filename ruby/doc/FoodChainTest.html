def test_read_guard
  song_file = File.expand_path('song.txt', __dir__)
  ["IO.read '#{song_file}'", "File.open '#{song_file}'"].each do |trigger|
    assert_raises(NoCheating) { FoodChain.send :class_eval, trigger }
  end
end
